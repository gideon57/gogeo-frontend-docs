# 图表

baseApp封装了 echarts (opens new window) 提供给appkit使用

使用 goChart 组件，并传入 宽 , 高 , option, 即可渲染 示例如下

  <template>
    <goChart width="300px" height="200px" :option="chartOption" />
  </template>
  <script>
    export default {
      data() {
        return {
          chartOption: {
            xAxis: {},
            yAxis: {},
            series: [
              {
                symbolSize: 20,
                data: [
                  [10.0, 8.04],
                  [8.07, 6.95],
                  [13.0, 7.58],
                  [9.05, 8.81],
                ],
                type: 'scatter'
              }
            ]
          }
        };
      },
    }
  </script>

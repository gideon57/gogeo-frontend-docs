# 事件

# 地图初始化事件

地图的 下钻初始化完成 地图容器初始化完成 图层列表初始化完成 事件

  <script>
    window.addEventListener("message", function (e) {
      let { eventName, prop } = e.data
      if (eventName === 'onDrillConfigInitReady') {
        console.log('下钻初始化完成')
      }else if(eventName === 'onMapContainerInitReady'){
        console.log('地图容器初始化完成')
      }else if(eventName === 'onLayerListInitReady'){
        console.log('图层列表初始化完成')
      }
    }, false);
  </script>
  <div id="micro-app-proMap">
    <iframe src="publishUrl" id="myIframe" frameborder="0"></iframe>
  </div>

# 获取所有图层图例信息

当图层获取成功、呈现设置改变、分段配置改变，会重新生成图例，地图会抛出事件，在appkit中可以获取相应信息

示例如下

  <script>
    window.addEventListener("message", function (e) {
      let { eventName, prop } = e.data
     if (eventName === 'getLegendFromPublish') {
        console.log('getLegendFromPublish', prop)
      }
    }, false);
  </script>
  <div id="micro-app-proMap">
    <iframe src="publishUrl" id="myIframe" frameborder="0"></iframe>
  </div>

# 刷新指定图层

appkit需要在适当的时机刷新指定图层，只需传入参数：layerCode

示例如下

  <script>
    function postMsg(visibleSwitch){
      let myIframe = document.querySelector('#myIframe')
      let props = {
        layerCode:'c73c8e8ee7f04631ad69461e445612fb'
      }
      myIframe.contentWindow.postMessage({eventName:'refreshLayerFromPublishMap',props}, '*')
    }
  </script>
  <div id="micro-app-proMap">
    <iframe src="publishUrl" id="myIframe" frameborder="0">
    </iframe>
    <button onclick="postMsg()" >刷新图层</button>
  </div>

# 过滤显示

根据layercode和点线面code过滤显示

假设有三个图层

需求共有三种可能：图层1全部显示不受影响，图层2全部隐藏，图层3根据code显示

则layerList为：图层1不传，图层2的codes为空数组，图层3的codes为需要显示的code集合

示例如下：

  <script>
    function postMsg(visibleSwitch){
      let myIframe = document.querySelector('#myIframe')
      let props = {
        switch:visibleSwitch,  //是否打开  1打开 0关闭
        layerList: [
          {
            layerCode: 'adb2e30251d74291b79b8fe7d720924f',
            codes:[],
          },
          {
            layerCode: '82cd310b352044d59fa59f0cb05fd35f',
            codes: ["72ce3eb6a2cd4b66b05f4cdaad91af3d"]
          }
        ]
      }
      myIframe.contentWindow.postMessage({eventName:'updateFeatureVisibleCodeFromPublishMap',props}, '*')
    }
  </script>
  <div id="micro-app-proMap">
    <iframe src="publishUrl" id="myIframe" frameborder="0">
    </iframe>
    <button onclick="postMsg(1)" >打开</button>
    <button onclick="postMsg(0)" >关闭</button>
  </div>

# 钻取

# appkit控制地图的下钻事件

   <script>
    function postMsg(drillType, layerCode, content, center){
      let myIframe = document.querySelector('#myIframe')
      let props = {
        drillType, layerCode, content, center
      }
      myIframe.contentWindow.postMessage({eventName:'setDrillByPublish',props}, '*')
    }
  </script>
  <div id="micro-app-proMap">
    <iframe src="publishUrl" id="myIframe" frameborder="0"></iframe>
    <button onclick="postMsg('move',null,'moveContent',[121.67787616994319,30.494136005203597])" >move</button>
    <button onclick="postMsg('out')" >out</button>
    <button onclick="postMsg('home')" >home</button>
    <button onclick="postMsg('in','ade9e9b3622c44b59501bb7734ce1fc9','inContent',[116.41671826194344,39.91291576311387])" >in</button>
  </div>

# 配置属性如下

以省、市、区三个图层距离

参数 说明 类型 备注 是否必填
drillType 下钻类型 string 取消市: out; 选择区：in； 切换市：move 是
layerCode 图层code string 选择区时，layerCode为市图层的layerCode drillType=in时必填
content 选中值 string 选择区时，content="朝阳区" 否
center 地图跳转中心点经纬度 Array [lng,lat] 经度在前纬度在后，如：[116.416718,39.912915] 否

# 控制分段显示隐藏

地图开启分段后，可在appkit中控制地图，按分段结果隐藏某一组数据

 <script>
    function postMsg(){
      let myIframe = document.querySelector('#myIframe')
      let props = {
        layerCode: '73683bdf828244b8b7b6f821a90df03d', sectionIndex: 0, visible: false
      }
      myIframe.contentWindow.postMessage({eventName:'switchSectionVisibleFromLegend',props}, '*')
    }
  </script>
  <div id="micro-app-proMap">
    <iframe src="publishUrl" id="myIframe" frameborder="0"></iframe>
    <button onclick="postMsg()" >发送</button>
  </div>

# 追加筛选条件

地图本身可配置筛选器挂件，appkit也可实时追加筛选条件，条件改变后地图会实时查询最新数据

提示： 该追加为覆盖追加，如要清空已追加的筛选条件，直接传空即可

   <script>

    function postMsg(){
      let props = [
          {
              "dataCode": "46b76e4afe034be7be288b8e39b023ff",
              "fieldName": "field42",
              "operationType": "eq",
              "content": "销量优先"
          },
          {
              "dataCode": "46b76e4afe034be7be288b8e39b023ff",
              "fieldName": "field39",
              "operationType": "eq",
              "content": "销量优先1"
          },
      ]
      let myIframe = document.querySelector('#myIframe')
      myIframe.contentWindow.postMessage({eventName:'setFilterFromPublishMap',props}, '*')
    }
  </script>
  <div id="micro-app-proMap">
    <iframe src="publishUrl" id="myIframe" frameborder="0"></iframe>
    <button onclick="postMsg()" >发送</button>
  </div>
